Connection Manager

  • Critical software component
    •
  • Decides which radio to connect to and when
    •
  • v1.0 implemented simple behaviour:
    •
    1. Scan for track-side access points (APs)
    2. Connect to AP with the strongest signal
    3. Wait for disconnection (and go back to 1)

The Lua Stack

  • Lua communicates with C++ through an API
    •
  • Values are passed between Lua and C++ using stacks
    •
  • API functions do not enforce strict stack discipline
    •
    • positive indices represent absolute positions
    • negative indices are relative to the top of the stack

Calling Lua from C++


a = f("how", t.x, 14)      -- Lua code to run in C++
            

lua_getglobal(L, "f");     /* function to be called */
lua_pushliteral(L, "how"); /* 1st argument */
lua_getglobal(L, "t");     /* table to be indexed */
lua_getfield(L, -1, "x");  /* push t.x (2nd arg) */
lua_remove(L, -2);         /* remove 't' from the stack */
lua_pushinteger(L, 14);    /* 3rd argument */
lua_call(L, 3, 1);         /* call 'f' (3 args, 1 res) */
lua_setglobal(L, "a");     /* set global 'a' */

Simple Example: C++


#include <iostream>

#include <lua.h>
#include <lauxlib.h>
#include <lualib.h>

int cppFunc(lua_State *L) {
  std::cout << "cppFunc called with args:" << std::endl;

  for (int n=1; n <= lua_gettop(L); ++n)
    std::cout << lua_tostring(L, n) << std::endl;

  return 0;
}

int main([[maybe_unused]] int argc, char ** argv)
{
  // Create a new lua state
  lua_State *L = luaL_newstate();

  // Open all libraries
  luaL_openlibs(L);

  // export a C++ function to Lua
  lua_register(L, "cppFunc", cppFunc);

  // Load and run the Lua file
  luaL_loadfile(L, argv[1]);
  lua_pcall(L, 0, LUA_MULTRET, 0);

  // Call 'f' with the arguments "how", t.x, 14
  lua_getglobal(L, "f");     /* function to be called */
  lua_pushliteral(L, "how"); /* 1st argument */
  lua_getglobal(L, "t");     /* table to be indexed */
  lua_getfield(L, -1, "x");  /* push t.x (2nd arg) */
  lua_remove(L, -2);         /* remove 't' from the stack */
  lua_pushinteger(L, 14);    /* 3rd argument */
  lua_call(L, 3, 1);         /* call 'f' (3 args, 1 res) */
  lua_setglobal(L, "a");     /* set global 'a' */

  lua_close(L);
}

LuaChat Arguments


void LuaManager::ProcessArguments(
  std::vector<std::string> const& arguments)
{
  lua_createtable(lua_state, 0, arguments.size());

  using namespace std::regex_constants
  std::regex number_regex(R"((\w+)=([-+]?[0-9]+\.?[0-9]*)$)");
  std::regex bool_regex(R"((\w+)=(true|false))", icase);
  std::regex string_regex(R"((\w+)=([[:graph:]]+))");

  for (auto const& arg : arguments) {
    std::smatch match;

    try {
      if (std::regex_search(arg, match, number_regex))
        lua_pushnumber(lua_state, std::stod(match[2].str()));
      else if (std::regex_search(arg, match, bool_regex))
        lua_pushboolean(lua_state,
          !strcasecmp(match[2].str().c_str(), "true")
        );
      else if (std::regex_search(arg, match, string_regex))
        lua_pushstring(lua_state, match[2].str().c_str());
      else
        throw std::runtime_error("Could not parse arg: " + arg);
    } catch (std::logic_error const& e) {
      throw std::runtime_error("Exception converting Lua argument "
                               + arg + ": " + e.what());
    }

    // Set the name of the key
    lua_setfield(lua_state, -2, match[1].str().c_str());
  }

  // Store a reference to the table for when the behaviour is run
  lua_ref_argument_table = luaL_ref(lua_state, LUA_REGISTRYINDEX);
  if (lua_ref_argument_table == LUA_REFNIL)
    throw std::runtime_error("Could not set Lua table reference");
}

SWIG

  • Simplified Wrapper and Interface Generator
    •
  • Produces C++ bindings for many target languages
    •
    • Generates Lua stack calls for std C++ types
    • std::string, std::vector, std::map etc.
  • Integrates well with CMake
    •

LuaChat SWIG CMake


set(LUA_CHAT_SWIG_SRCS
    lua_chat_actions.i lua_chat_action_log.cpp)

set_source_files_properties(${LUA_CHAT_SWIG_SRCS}
                            PROPERTIES CPLUSPLUS ON)

swig_add_library(actions TYPE USE_BUILD_SHARED_LIBS
                 LANGUAGE lua
                 SOURCES ${LUA_CHAT_SWIG_SRCS})

target_include_directories(actions PRIVATE
                           ${CMAKE_CURRENT_SOURCE_DIR}
                           ${LUA_CHAT_SOURCE_DIR}/src
                           ${LUA_CHAT_SOURCE_DIR}/third_party
                           ${LUA_CHAT_SOURCE_DIR}/third_party/asio
                           ${LUA_INCLUDE_DIR})

target_compile_definitions(actions PRIVATE ASIO_STANDALONE)

target_link_libraries(actions PRIVATE std::filesystem)
