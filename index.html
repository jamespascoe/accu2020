The Lua Stack

  • Lua communicates with C++ through an API
    •
  • Values are passed between Lua and C++ using stacks
    •
  • C++ API does not enforce strict stack discipline
    •
    • positive indices represent absolute positions
    • negative indices are relative to the top of the stack

Calling Lua from C++


a = f("how", t.x, 14)      -- Lua code to run in C++
            

lua_getglobal(L, "f");     /* function to be called */
lua_pushliteral(L, "how"); /* 1st argument */
lua_getglobal(L, "t");     /* table to be indexed */
lua_getfield(L, -1, "x");  /* push t.x (2nd arg) */
lua_remove(L, -2);         /* remove 't' from the stack */
lua_pushinteger(L, 14);    /* 3rd argument */
lua_call(L, 3, 1);         /* call 'f' (3 args, 1 res) */
lua_setglobal(L, "a");     /* set global 'a' */

Simple Example: C++


#include <iostream>

#include <lua.hpp>

int cppFunc(lua_State *L) {
  std::cout << "cppFunc called with args:" << std::endl;

  for (int n=1; n <= lua_gettop(L); ++n)
    std::cout << lua_tostring(L, n) << std::endl;

  return 0;
}

int main([[maybe_unused]] int argc, char ** argv)
{
  // Create a new lua state
  lua_State *L = luaL_newstate();

  // Open all libraries
  luaL_openlibs(L);

  // export a C++ function to Lua
  lua_register(L, "cppFunc", cppFunc);

  // Load and run the Lua file
  luaL_loadfile(L, argv[1]);
  lua_pcall(L, 0, LUA_MULTRET, 0);

  // Call 'f' with the arguments "how", t.x, 14
  lua_getglobal(L, "f");     /* function to be called */
  lua_pushliteral(L, "how"); /* 1st argument */
  lua_getglobal(L, "t");     /* table to be indexed */
  lua_getfield(L, -1, "x");  /* push t.x (2nd arg) */
  lua_remove(L, -2);         /* remove 't' from the stack */
  lua_pushinteger(L, 14);    /* 3rd argument */
  lua_call(L, 3, 1);         /* call 'f' (3 args, 1 res) */
  lua_setglobal(L, "a");     /* set global 'a' */

  lua_close(L);
}

LuaChat Arguments


void LuaManager::ProcessArguments(
  std::vector<std::string> const& arguments)
{
  lua_createtable(lua_state, 0, arguments.size());

  using namespace std::regex_constants
  std::regex number_regex(R"((\w+)=([-+]?[0-9]+\.?[0-9]*)$)");
  std::regex bool_regex(R"((\w+)=(true|false))", icase);
  std::regex string_regex(R"((\w+)=([[:graph:]]+))");

  for (auto const& arg : arguments) {
    std::smatch match;

    try {
      if (std::regex_search(arg, match, number_regex))
        lua_pushnumber(lua_state, std::stod(match[2].str()));
      else if (std::regex_search(arg, match, bool_regex))
        lua_pushboolean(lua_state,
          !strcasecmp(match[2].str().c_str(), "true")
        );
      else if (std::regex_search(arg, match, string_regex))
        lua_pushstring(lua_state, match[2].str().c_str());
      else
        throw std::runtime_error("Could not parse arg: " + arg);
    } catch (std::logic_error const& e) {
      throw std::runtime_error("Exception converting argument "
                               + arg + ": " + e.what());
    }

    // Set the name of the key
    lua_setfield(lua_state, -2, match[1].str().c_str());
  }

  // Store a reference to the table for when the behaviour is run
  lua_ref_argument_table = luaL_ref(lua_state, LUA_REGISTRYINDEX);
  if (lua_ref_argument_table == LUA_REFNIL)
    throw std::runtime_error("Could not set Lua table reference");
}

SWIG

  • Simplified Wrapper and Interface Generator
    •
  • Produces C++ bindings for many target languages
    •
    • Generates Lua stack calls for std C++ types
    • std::string, std::vector, std::map etc.
  • Integrates well with CMake

LuaChat SWIG CMake


set(LUA_CHAT_SWIG_SRCS
    lua_chat_actions.i lua_chat_action_log.cpp)

set_source_files_properties(${LUA_CHAT_SWIG_SRCS}
                            PROPERTIES CPLUSPLUS ON)

swig_add_library(actions TYPE USE_BUILD_SHARED_LIBS
                 LANGUAGE lua
                 SOURCES ${LUA_CHAT_SWIG_SRCS})

target_include_directories(actions PRIVATE
                           ${CMAKE_CURRENT_SOURCE_DIR}
                           ${LUA_CHAT_SOURCE_DIR}/src
                           ${LUA_CHAT_SOURCE_DIR}/third_party
                           ${LUA_INCLUDE_DIR})

target_compile_definitions(actions PRIVATE ASIO_STANDALONE)

target_link_libraries(actions PRIVATE std::filesystem)

SWIG Typemaps

Lua Callback: SWIG Input


%include <lua_fnptr.i>

%typemap(typecheck) Example::Callback & {
  $1 = lua_isfunction(L, $input);
}

%typemap(in) Example::Callback & (Example::Callback temp) {
  // Create a reference to the Lua callback
  SWIGLUA_REF fn;
  swiglua_ref_set(&fn, L, $input);

  temp = [&fn]() {
    swiglua_ref_get(&fn);

    lua_pcall(fn.L, 0, 0, 0);
  };

  $1 = &temp;
}

Actions

LuaChat Actions

  • Talk: sends messages to a remote LuaChat
    •
    • Based on Asio - must also act as a server
    • Use TCP for fault-tolerant in-order delivery
    • One asynchronous TCP connection per message
  • Timer: implements blocking and non-blocking waits
    •
    • Use Asio - required for Lua coroutine scheduler
  • Log: wraps spdlog primitives

TCP Connections


class tcp_connection
  : public std::enable_shared_from_this<tcp_connection> {
public:
  using pointer = std::shared_ptr<tcp_connection>

  static pointer create(asio::io_context& io_context) {
    return pointer(new tcp_connection(io_context));
  }

  asio::ip::tcp::socket& socket() { return m_socket; }

  std::string& data() { return m_data; }

private:
  tcp_connection(asio::io_context& io_context)
    : m_socket(io_context) {}

  asio::ip::tcp::socket m_socket;
  std::string m_data;
};

Connection Handling


Talk::Talk(unsigned short port)
    : m_acceptor(m_io_context,
                tcp::endpoint(tcp::v4(), port)) {
  start_accept();

  m_thread = std::thread([this](){ m_io_context.run(); });

  log_trace("Talk action starting");
}

void Talk::start_accept() {
  tcp_connection::pointer connection =
      tcp_connection::create(
        m_acceptor.get_executor().context()
      );

  m_acceptor.async_accept(connection->socket(),
      [this, connection](const asio::error_code& error) {
        handle_accept(connection, error);
      }
  );
}

Storing Data


void Talk::handle_read(asio::error_code const& error,
                       std::size_t bytes_transferred,
                       tcp_connection::pointer connection) {
  if (!error || error == asio::error::eof) {
    // Limit the message array
    if (m_messages.size() > max_messages)
      m_messages.erase(m_messages.begin());

    m_messages.emplace_back(connection->data());

    log_info("Received message ({} bytes): {}",
             bytes_transferred,
             connection->data());
  } else
    log_error("Talk read failed: {}", error.message());
}

Lua Retrieval


std::string Talk::GetNextMessage(void) {
  if (!IsMessageAvailable())
    return "";

  std::string ret = m_messages.front();

  m_messages.erase(m_messages.begin());

  return ret;
}

Behaviour

Coroutines

  • Stackful (fibers): coroutine state includes call stack
    •
    • Examples: Lua, Boost.Coroutine
      •
    • pros: powerful, nested suspend and resume
      •
    • cons: memory intensive
      •
  • Stackless: state does not include call stack
    •
    • Examples: C++20, co2, JavaScript
      •
    • pros: massively scalable, faster context switch
      •
    • cons: only the coroutine top-level may suspend
      •

Lua Concurrency

  • Coroutines: Multithreaded C++ and Lua Coroutines
    •
  • LuaLanes: OS threads with asynchronous messaging
  • luaproc: official tool from Lua development team
    •
  • Effil: clean API, shared tables, good performance
    •
  • ZeroMQ: bindings for C++, Lua and FFI
    •
  • mtmsg + mtstates + mtint: building blocks

Lua Coroutines

  • Similar to a thread, but not pre-emptive
    •
  • Brilliant for eliminating latency
    •
  • Single threaded so lock-free, no races etc.
    •
  • Implement your own scheduler
    •

LuaChat Behaviour

  • Sender coroutine: sends user input to peer
    •
  • Receiver coroutine: prints received messages
    •
  • Dispatcher coroutine: schedules sender and receiver
    •
  • main: processes arguments and creates coroutines
    •

Sender Coroutine


function sender (talk, host, port)

  while true do

    local ret = require 'posix'.rpoll(0, 1000)
    if (ret == 1) then
      local message = io.read()
      if (message ~= "") then

        local ret = talk:Send(
          tostring(host), tostring(port), tostring(message)
        )

        if (ret == Actions.Talk.ErrorType_SUCCESS) then
          Actions.Log.info(
            string.format(
              "Message sent to %s:%s %s", host, port, message
            )
          )
        end
      end
    end

    coroutine.yield()

  end

end

Receiver Coroutine


function receiver (talk, host, port)

  while true do

    -- Yield until a message arrives, at which point, print it
    repeat
      coroutine.yield()
    until talk:IsMessageAvailable()

    local message = talk:GetNextMessage()

    Actions.Log.info(
      string.format(
        "Received from %s:%s %s", host, port, message
      )
    )

    print(host .. ":" .. tostring(port) .. "> " .. message)

  end

end

Dispatcher


function dispatcher (coroutines)

  local timer = Actions.Timer()

  while true do
    if next(coroutines) == nil then break end -- no coroutines

    for name, co in pairs(coroutines) do
      local status, res = coroutine.resume(co)

      if res then -- coroutine has exited

        if type(res) == "string" then -- runtime error
          Actions.Log.critical(
            "Coroutine '" .. tostring(name) .. "' error " .. res
          )
        else
          Actions.Log.warn(
            "Coroutine '" .. tostring(name) .. "' exited"
          )
        end

        coroutines[name] = nil
      end
    end

    timer(Actions.Timer.WaitType_BLOCK, 1, "ms", 0xffffffff)

  end
end

Performance

  • Cavium Octeon ThunderX CN81 NPU @ 1.5Ghz
    •
    • 4 ARMv8 cores, 78KB I$, 32KB D$, 2MB L2, CCI

  • MCM is not compute or IO bound
    •
  • But we do want to minimise event response time
    •
    • E.g. handover, switching links, decision making

Advice

  • MCM spends a lot of time in the SWIG wrapper
    •
    • prefer lightweight typemaps
      •
  • The partition between C++ and Lua is important
    •
    • as is the concurrency design
      •
  • How the code interacts with Lua is significant
    •
    • prefer precompiled long-lived behaviours
      •
